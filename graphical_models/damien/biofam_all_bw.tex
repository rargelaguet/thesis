\begin{tikzpicture}
  % Define nodes:
  % matrix factorisation level
  \node[obs]   (Y) {$y_{n,d}^{m,g}$};
  \node[latent, above=of Y, xshift=-1.5cm] (Z) {$z^g_{n,k}$};
  \node[latent, above=of Y, xshift=1.5cm] (W) {$w_{k,d}^m$};
  \node[latent, densely dashed, xshift=1.5cm] (Tau) {$\tau_{d}^m$};
  \node[latent, densely dashed, yshift=-1.5cm] (TauN) {$\tau_{n}^{m,g}$};

  % \node[opacity=0.4,latent, xshift=-1.5cm] (Tau2) {$\tau_{n}$};

  % parents of Z
  \node[det, above=of Z] (crossZ) {$\times$};
  \node[latent, above=of crossZ] (Zhat) {$\hat{z}^g_{n,k}$};
  \node[latent, densely dotted, above=of Zhat] (SigmaZ) {$\alpha^g_k$};
  \node[latent, densely dotted, above=of crossZ, xshift=-1.5cm] (SZ) {$s^g_{n,k}$};
  \node[latent, densely dotted, above=of SZ] (ThetaZ) {$\theta^g_{k}$};

  % parents of W
  \node[det, above=of W] (crossW) {$\times$};
  \node[latent, above=of crossW] (What) {$\hat{w}_{k,d}^m$};
  \node[latent, densely dotted, above=of What] (SigmaW) {$\alpha_k^m$};
  \node[latent, densely dotted, above=of crossW, xshift=1.5cm] (SW) {$s_{k,d}^m$};
  \node[latent, densely dotted, above=of SW] (ThetaW) {$\theta_{k}^m$};

  % Connect the nodes
  \edge {Z,W} {Y}; %
  \edge[densely dashed] {Tau} {Y};
  \edge[densely dotted] {ThetaZ} {SZ};
  \edge[densely dotted] {SigmaZ} {Zhat};
  \edge[densely dotted] {ThetaW} {SW};
  \edge[densely dotted] {SigmaW} {What};
  \edge[densely dashed] {TauN} {Y}

  \factoredge[densely dotted] {SZ} {crossZ} {Z};
  \factoredge {Zhat} {crossZ} {Z};
  \factoredge[densely dotted] {SW} {crossW} {W};
  \factoredge {What} {crossW} {W};


  % Plates
  %factors
  \plate[] {plateK} {(Z)(W)(SZ)(Zhat)(SW)(What)(ThetaZ)(ThetaW)} {$K$};

  % feature side
  \plate[] {plateD} {(Y)(W)(Tau)(crossW)(What)(SW)(plateK.south east)} {$D_m$};
  \plate[] {plateM} {(TauN)(plateD)(ThetaW)(plateK.north east)} {$M$};

  % sample side
  \plate[] {plateN} {(TauN)(Y)(Z)(crossZ)(Zhat)(SZ)(plateK.south west)(plateD.south west)} {$N_g$};
  \plate[] {plateG} {(plateN)(plateK.north west)(plateM.north west)(plateM.north west)} {$G$};
\end{tikzpicture}
