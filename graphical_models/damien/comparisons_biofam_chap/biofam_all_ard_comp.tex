\begin{tikzpicture}
  % Define nodes:
  % matrix factorisation level
  \node[obs, ultra thick]   (Y) {$\mathbf{y_{n,d}^{m,g}}$};
  \node[latent, above=of Y, xshift=-1.5cm, ultra thick] (Z) {$\mathbf{z^g_{n,k}}$};
  \node[latent, above=of Y, xshift=1.5cm, ultra thick] (W) {$\mathbf{w_{k,d}^m}$};
  \node[latent, xshift=1.5cm, ultra thick] (Tau) {$\mathbf{\tau_{d}^m}$};
  \node[latent, yshift=-1.5cm, opacity=\op] (TauN) {$\tau_{n}^{m,g}$};

  % \node[opacity=0.4,latent, xshift=-1.5cm] (Tau2) {$\tau_{n}$};

  % parents of Z
  \node[det, above=of Z, ultra thick] (crossZ) {$\times$};
  \node[latent, above=of crossZ, ultra thick] (Zhat) {$\mathbf{\hat{z}^g_{n,k}}$};
  \node[latent, above=of Zhat, ultra thick] (SigmaZ) {$\mathbf{\alpha^g_k}$};
  \node[latent, above=of crossZ, xshift=-1.5cm, opacity=\op] (SZ) {$s^g_{n,k}$};
  \node[latent, above=of SZ, opacity=\op] (ThetaZ) {$\theta^g_{k}$};

  % parents of W
  \node[det, above=of W, ultra thick] (crossW) {$\times$};
  \node[latent, above=of crossW, ultra thick] (What) {$\mathbf{\hat{w}_{k,d}^m}$};
  \node[latent, above=of What, ultra thick] (SigmaW) {$\mathbf{\alpha_k^m}$};
  \node[latent, above=of crossW, xshift=1.5cm, opacity=\op] (SW) {$s_{k,d}^m$};
  \node[latent, above=of SW, opacity=\op] (ThetaW) {$\theta_{k}^m$};

  % Connect the nodes
  \edge[ultra thick] {Z,W} {Y}; %
  \edge[ultra thick] {Tau} {Y};
  \edge[opacity=\op] {ThetaZ} {SZ};
  \edge[ultra thick] {SigmaZ} {Zhat};
  \edge[opacity=\op] {ThetaW} {SW};
  \edge[ultra thick] {SigmaW} {What};
  \edge[opacity=\op] {TauN} {Y}

  \factoredge[opacity=\op] {SZ} {crossZ} {Z};
  \factoredge[ultra thick] {Zhat} {crossZ} {Z};
  \factoredge[opacity=\op] {SW} {crossW} {W};
  \factoredge[ultra thick] {What} {crossW} {W};


  % Plates
  %factors
  \plate[ultra thick] {plateK} {(Z)(W)(SZ)(Zhat)(SW)(What)(ThetaZ)(ThetaW)} {$\mathbf{K}$};

  % feature side
  \plate[ultra thick] {plateD} {(Y)(W)(Tau)(crossW)(What)(SW)(plateK.south east)} {$\mathbf{D_m}$};
  \plate[ultra thick] {plateM} {(TauN)(plateD)(ThetaW)(plateK.north east)} {$\mathbf{M}$};

  % sample side
  \plate[ultra thick] {plateN} {(TauN)(Y)(Z)(crossZ)(Zhat)(SZ)(plateK.south west)(plateD.south west)} {$\mathbf{N_g}$};
  \plate[ultra thick] {plateG} {(plateN)(plateK.north west)(plateM.north west)(plateM.north west)} {$\mathbf{G}$};
